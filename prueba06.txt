-- Función para calcular la secuencia de Fibonacci
entero funcion fibonacci(entero n)
{
    entero resultado = 0;

    si (n <= 0) 
    {
        resultado = 0;
    } 
    contrariosi (n == 1) 
    {
        resultado = 1;
    } 
    contrario 
    {
        entero a = 0;
        entero b = 1;

        para (entero i = 2 hastaque i <= n enpaso 1)
        {
            resultado = a + b;
            a = b;
            b = resultado;
        }
    }

    devolver resultado;
}


-- Función principal
entero funcion principal()
{
    -- Declaración de variables locales
    entero variable_local = 5;
    --entero variable_local;
    decimal variable_decimal = 3.14;
    texto variable_texto = "Hola, Leo!";
    boleano variable_booleana = verdadero;
    
    -- Estructuras de control
    si (variable_local > 0) 
    {
        imprimir("La variable_local es positiva");
    } 
    contrariosi (variable_local < 0) 
    {
        imprimir("La variable_local es negativa");
    } 
    contrario 
    {
        imprimir("La variable_local es cero");
    }

    mientras (variable_local > 0) 
    {
        imprimir("En el bucle mientras. Variable_local: ", variable_local);
        variable_local = variable_local - 1;
    }

    variable_local = 5;

    repetir 
    {
        imprimir("En el bucle repetir. Variable_local: ", variable_local);
        variable_local = variable_local - 1;
    } hastaque (variable_local <= 0)

    -- Operaciones aritméticas
    entero resultado_suma = variable_local + 10;
    entero resultado_resta = variable_local - 2;
    entero resultado_mult = variable_local * 3;
    entero resultado_div = variable_local / 2;

    -- Llamada a una función
    entero resultado_fibonacci = fibonacci(5);

    -- Imprimir resultados
    imprimir("Variable Decimal: ", variable_decimal);
    imprimir("Variable Texto: ", variable_texto);
    imprimir("Variable Booleana: ", variable_booleana);
    imprimir("Resultado Suma: ", resultado_suma);
    imprimir("Resultado Resta: ", resultado_resta);
    imprimir("Resultado Multiplicación: ", resultado_mult);
    imprimir("Resultado División: ", resultado_div);
    imprimir("Resultado de la función Fibonacci: ", resultado_fibonacci);

    -- Devolver un valor
    devolver 0;
}
